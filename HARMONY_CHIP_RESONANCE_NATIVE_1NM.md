              ellipsometry_data: Dict[str, float],   # Oxide thickness

        mu_purity = self._analyze_raman(raman_spectrum)
        mu_oxidation = self._analyze_oxide(ellipsometry_data)
        
        # Defect penalty
        mu_defect = math.exp(-defect_count / 1e10)  # 10^10 defects/cmÂ² threshold
        
        return (mu_surface * mu_interface * mu_purity * mu_oxidation * mu_defect) ** 0.2

        """Surface reconstruction quality from LEED"""
        if 'intensity' not in leed or 'k_x' not in leed:
            return 1.0
        
        # Sharp spots = ordered surface
        # Diffuse background = disorder
        spot_intensity = np.max(leed['intensity'])
        total_intensity = np.sum(leed['intensity'])
        
        if total_intensity == 0:
            return 0.0
        
        sharpness = spot_intensity / total_intensity
        
        # Expected spots for substrate/orientation
        expected_spots = self._get_expected_spots()
        detected_spots = self._count_spots(leed, threshold=0.1)
        
        spot_completeness = min(detected_spots / expected_spots, 1.0)
        
        return (sharpness * spot_completeness) ** 0.5
    
    def _analyze_xps(self, xps: Dict[str, float]) -> float:
        """Interface chemistry from XPS"""
        # Reference binding energies for pure materials
        ref_be = self._get_reference_binding_energies()
        
        shifts = []
        for element, measured in xps.items():
            if element in ref_be:
                shift = abs(measured - ref_be[element])
                shifts.append(shift)
        
        if not shifts:
            return 1.0
        
        max_shift = max(shifts)
        # 0.5 eV shift indicates chemical reaction/interface state
        return math.exp(-max_shift / 0.5)
    
    def _analyze_raman(self, raman: Dict) -> float:
        """Contamination detection from Raman"""
        # Look for Si-O, C-H, metal-oxide peaks
        contaminant_regions = [
            (200, 600),    # Metal oxides
            (900, 1100),   # Si-O-Si
            (2800, 3000),  # C-H stretch
        ]
        
        if 'wavenumber' not in raman or 'intensity' not in raman:
            return 1.0
        
        contamination = 0
        for low, high in contaminant_regions:
            mask = (raman['wavenumber'] >= low) & (raman['wavenumber'] <= high)
            contamination += np.sum(raman['intensity'][mask])
        
        total = np.sum(raman['intensity'])
        if total == 0:
            return 1.0
        
        ratio = contamination / total
        return max(0, 1 - ratio * 10)  # 10% contamination = 0 score

            return 1.0
        
        t_oxide = ellipsometry['oxide_thickness_nm']

        return math.exp(-t_oxide / t_critical)

            return A * np.exp(-(x - x0)**2 / (2 * sigma**2))
